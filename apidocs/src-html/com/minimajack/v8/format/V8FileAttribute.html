001package com.minimajack.v8.format;
002
003import com.minimajack.v8.utils.DateUtils;
004
005import java.io.DataOutput;
006import java.io.IOException;
007import java.nio.ByteBuffer;
008import java.nio.charset.Charset;
009import java.util.Date;
010
011public class V8FileAttribute extends BlockHeader {
012
013        public static final int ATTIBUTES_SIZE = 24; // 2x8 + 1x4 + 1x4
014
015        public static final int MINIMUM_PAYLOAD_SIZE = 32; // 2x8 + 1x4 + 1x4
016
017        public static final Charset DEFAULT_CHARSET = Charset.forName("UTF-16LE");
018
019        private long creationDate;
020
021        private long modifyDate;
022
023        private String name;
024
025        private int attributes;
026
027        public V8FileAttribute() {
028                super();
029        }
030
031        public V8FileAttribute(ByteBuffer buffer, int position) {
032                super(buffer, position);
033        }
034
035        @Override
036        public void read() throws IOException {
037                this.readHeader();
038                this.creationDate = getBuffer().getLong();
039                this.modifyDate = getBuffer().getLong();
040                getBuffer().getInt(); // reserved
041
042                byte[] stringArray = new byte[this.getDocSize() - ATTIBUTES_SIZE];
043                this.getBuffer().get(stringArray);
044                V8FileAttribute.this.name = new String(stringArray, DEFAULT_CHARSET);
045                this.attributes = getBuffer().getInt();
046        }
047
048        @Override
049        public void write(DataOutput buffer) throws IOException {
050                int payloadSize = getPayloadSize();
051
052                this.setDocSize(payloadSize);
053
054                buffer.writeLong(creationDate);
055                buffer.writeLong(modifyDate);
056                buffer.writeInt(0);
057                buffer.write(name.getBytes(DEFAULT_CHARSET));
058                buffer.writeInt(attributes);
059                buffer.write(new byte[this.getBlockSize() - payloadSize]);
060        }
061
062        @Override
063        public void setDocSize(int docSize) {
064                super.setDocSize(Math.max(MINIMUM_PAYLOAD_SIZE, docSize));
065        }
066
067        public int getPayloadSize() {
068                int payloadSize = ATTIBUTES_SIZE;
069                byte[] names = name.getBytes(DEFAULT_CHARSET);
070                payloadSize += names.length;
071                return payloadSize;
072        }
073
074        public Date getCreationDate() {
075                return DateUtils.fromv8Time(this.creationDate);
076        }
077        
078    public Date safeGetCreationDate() {
079        Date creationDate =getCreationDate();
080        if ( creationDate == null )
081        {
082            return getModifyDate();
083        }else{
084            return creationDate;
085        }
086    }
087    
088        public void setCreationDate(Date creationDate) {
089                this.creationDate = DateUtils.tov8Time(creationDate);
090        }
091
092        public Date getModifyDate() {
093                return DateUtils.fromv8Time(this.modifyDate);
094        }
095
096        public void setModifyDate(Date modifyDate) {
097                this.modifyDate = DateUtils.tov8Time(modifyDate);
098        }
099
100        public int getAttributes() {
101                return attributes;
102        }
103
104        public void setAttributes(int attributes) {
105                this.attributes = attributes;
106        }
107
108        public String getName() {
109                return name;
110        }
111
112        public void setName(String name) {
113                this.name = name;
114        }
115
116}
